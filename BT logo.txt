    <script>
        function openBatch(evt, batchName) {
            var i, batchSections, tablinks;
            batchSections = document.getElementsByClassName("batch");
            for (i = 0; i < batchSections.length; i++) {
                batchSections[i].style.display = "none";
            }
            tablinks = document.getElementsByClassName("tablink");
            for (i = 0; i < tablinks.length; i++) {
                tablinks[i].className = tablinks[i].className.replace(" active", "");
            }
            document.getElementById(batchName).style.display = "block";
            evt.currentTarget.className += " active";
        }
        function toggleChapter(evt, dropID) {
            // Hide all dropdowns first
            document.querySelectorAll('.dropdown-content').forEach(function(el){
                if(el.id !== dropID) el.style.display = "none";
            });
            var dropMenu = document.getElementById(dropID);
            if (dropMenu.style.display === "block") {
                dropMenu.style.display = "none";
            } else {
                dropMenu.style.display = "block";
            }
            evt.stopPropagation();
        }
        // Hide dropdown if clicking elsewhere
        document.addEventListener('click', function(e){
            document.querySelectorAll('.dropdown-content').forEach(function(d){
                d.style.display = "none";
            });
        });
        // Prevent closing when clicking inside dropdown
        document.querySelectorAll('.chapter-tab').forEach(function(tab){
            tab.addEventListener('click', function(e){ e.stopPropagation(); });
        });
    </script>
