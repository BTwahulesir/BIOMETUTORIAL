        function openBatch(evt, batchName) {
            var i, batchSections, tablinks;
            batchSections = document.getElementsByClassName("batch");
            for (i = 0; i < batchSections.length; i++) {
                batchSections[i].style.display = "none";
            }
            tablinks = document.getElementsByClassName("tablink");
            for (i = 0; i < tablinks.length; i++) {
                tablinks[i].className = tablinks[i].className.replace(" active", "");
            }
            document.getElementById(batchName).style.display = "block";
            evt.currentTarget.className += " active";
        }
    </script>
